import ray

@ray.remote
class Counter:
    def __init__(self):
        self.value = 0
    def increment(self):
        self.value += 1
        return self.value
    def get_counter(self):
        return self.value

counters = [Counter.remote() for _ in range(10)]

results = ray.get([c.increment.remote() for c in counters])

results = ray.get([counters[0].increment.remote() for _ in range(5)])
